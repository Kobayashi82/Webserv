
<?php
// Establecer la cabecera para indicar que el contenido es texto plano
header('Content-Type: text/plain; charset=UTF-8');

// Leer el contenido del cuerpo de la solicitud
$bodyContent = file_get_contents('php://input');

// Imprimir el contenido de los datos POST (contenido del formulario)
echo "Datos recibidos en \$_POST:\n";
if (!empty($_POST)) {
    print_r($_POST);
} else {
    echo "No se recibieron datos en \$_POST.\n";
}

// Imprimir el contenido del cuerpo de la solicitud
echo "\nCuerpo recibido desde php://input:\n";
if (!empty($bodyContent)) {
    echo $bodyContent;
} else {
    echo "No se recibió ningún dato en el cuerpo del mensaje.\n";
}

// También podemos imprimir los encabezados de la solicitud para ver más detalles
echo "\nEncabezados de la solicitud:\n";
foreach (getallheaders() as $name => $value) {
    echo "$name: $value\n";
}
